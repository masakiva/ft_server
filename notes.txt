download from docker-hub: 		docker pull centos
run docker image: 				docker run -t --name acertainname centos (-d to run in bg)
map container expose port to local port: -p host:container (e.g. 8080:80 to map http port of nginx to localhost:8080)
sync a folder in the container to a local folder: -v host:container (e.g. /home/ah/42/ft_server/src:/usr/share/nginx/html:ro with ro for read-only)
logs of a running container:	docker logs acertainname (-f to leave open)
status of running containers: 	docker ps (-a to see stopped ones)
with statistics:				docker stats
connect to centos shell:		docker exec -it acertainname bash
stop docker container:			docker stop acertainname
start stopped container:		docker start acertainname
remove container:				docker rm acertainname

run & connect to alpine shell:	docker run --rm -ti alpine-smarter:1.0 /bin/sh
list intermediate build images:	docker images -a (Only RUN, COPY and ADD instructions create layers to improve build performance.)

build img from dockerfile dir:	docker build -t debian-server:1.0 .
list dangling images:			docker images --filter "dangling=true"


run nginx server:				nginx (or service nginx start)
stop nginx server:				nginx -s quit
check if it is running:			service nginx status

start mariadb/mysql:			service mysql start

dl phpmyadmin:					cd /tmp
								wget https://files.phpmyadmin.net/phpMyAdmin/5.0.2/phpMyAdmin-5.0.2-english.tar.gz
								tar -xzf phpMyAdmin-5.0.2-english.tar.gz
								mv phpMyAdmin-5.0.2-english /usr/share/phpmyadmin
